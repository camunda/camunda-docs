Evaluates a FEEL expression and returns the result. Supports references to tenant scoped cluster variables when a tenant ID is provided.

<ParamsDetails parameters={undefined}></ParamsDetails>

<RequestSchema
  title={"Body"}
  body={{
    required: true,
    content: {
      "application/json": {
        schema: {
          type: "object",
          required: ["expression"],
          properties: {
            expression: {
              type: "string",
              description: 'The expression to evaluate (e.g., "=x + y")',
              example: "=x + y",
            },
            tenantId: {
              type: "string",
              description:
                "Required when the expression references tenant-scoped cluster variables",
              example: "tenant_123",
            },
            variables: {
              type: "object",
              description:
                "Optional variables for expression evaluation. These variables are only used for the current evaluation and do not persist beyond it.",
              example: { x: 10, y: 20 },
              additionalProperties: true,
              nullable: true,
            },
          },
          title: "ExpressionEvaluationRequest",
        },
      },
    },
  }}
