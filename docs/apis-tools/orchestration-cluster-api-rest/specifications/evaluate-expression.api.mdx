Evaluates a FEEL expression and returns the result. Supports references to tenant scoped cluster variables when a tenant ID is provided.

    content: {
      "application/json": {
        schema: {
          type: "object",
          required: ["expression"],
          properties: {
            expression: {
              type: "string",
              description: 'The expression to evaluate (e.g., "=x + y")',
              example: "=x + y",
            },
            tenantId: {
              type: "string",
              description:
                "Required when the expression references tenant-scoped cluster variables",
              example: "tenant_123",
            },
            variables: {
              type: "object",
